            config:
              issuer_url: https://accounts.google.com
              client_id: <your-oidc-client-id>
              client_secret: <your-oidc-client-secret>
              scopes:
                - openid
                - profile
                - email
      - name: Headers
        actions:
          - type: add-headers
            config:
              headers:
                id-token: ${actions.ngrok.oidc.identity_token}
