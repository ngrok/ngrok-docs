```yaml policy.yml
on_http_request:
  - actions:
      - type: jwt-validation
        config:
          issuer:
            allow_list:
              - value: https://<YOUR_AUTH0_TENANT_ID>.<AUTH0_REGION>.auth0.com/
          audience:
            allow_list:
              - value: https://<YOUR_NGROK_DOMAIN>
          http:
            tokens:
              - type: jwt
                method: header
                name: Authorization
                prefix: 'Bearer '
          jws:
            allowed_algorithms:
              - RS256
            keys:
              sources:
                additional_jkus:
                  - https://<YOUR_AUTH0_TENANT_ID>.<AUTH0_REGION>.auth0.com/.well-known/jwks.json
```
