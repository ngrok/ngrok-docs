```go
import (
	"context"
	"net"

	"golang.ngrok.com/ngrok/v2"
)

func ngrokListener(ctx context.Context) (net.Listener, error) {
	return ngrok.Forward(ctx,
		ngrok.WithUpstream("tcp://localhost:8080",
			ngrok.WithUpstreamProxyProto(ngrok.ProxyProtoV2),
		),
		ngrok.WithURL("tcp://"),
	)
}
```
