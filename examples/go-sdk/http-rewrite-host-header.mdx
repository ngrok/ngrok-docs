```go
import (
	"context"
	"net"

	"golang.ngrok.com/ngrok/v2"
)

trafficPolicy := `
on_http_request:
  - actions:
      - type: add-headers
        config:
          headers:
            host: localhost
`

func ngrokListener(ctx context.Context) (net.Listener, error) {
	return ngrok.Listen(ctx,
		ngrok.WithTrafficPolicy(trafficPolicy),
	)
}
```
