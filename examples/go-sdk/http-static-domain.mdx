```go
import (
	"context"
	"net"

	"golang.ngrok.com/ngrok"
	"golang.ngrok.com/ngrok/config"
)

func ngrokListener(ctx context.Context) (net.Listener, error) {
	return ngrok.Listen(ctx,
		config.HTTPEndpoint(
                        config.WithDomain("example.ngrok.app"),
                ),
		ngrok.WithAuthtokenFromEnv(),
	)
}
```
