```go
import (
	"context"
	"net"

	"golang.ngrok.com/ngrok/v2"
)

func ngrokListener(ctx context.Context) (net.Listener, error) {
	return ngrok.Forward(ctx,
		ngrok.WithUpstream("http://localhost:8080"),
		ngrok.WithURL("https://example.ngrok.app"),
	)
}
```
