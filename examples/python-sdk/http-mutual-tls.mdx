```python
import ngrok

def load_file(name):
    with open(name, "r") as crt:
        return bytearray(crt.read().encode())

listener = ngrok.connect("localhost:8080", authtoken_from_env=True,
	mutual_tls_cas=load_file("/path/to/cas.pem"))

print(f"Ingress established at: {listener.url()}");
```
