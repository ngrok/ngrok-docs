```python
import ngrok

listener = ngrok.forward("localhost:8080", authtoken_from_env=True,
    proto="tls",
    ip_restriction_allow_cidrs=["110.0.0.0/8", "220.12.0.0/16"],
    ip_restriction_deny_cidrs="110.2.3.4/32")

print(f"Ingress established at: {listener.url()}");
```
