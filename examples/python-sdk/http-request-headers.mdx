```python
import ngrok

listener = ngrok.connect("localhost:8080", authtoken_from_env=True,
    request_header_add=["is-ngrok:1", "country:${.ngrok.geo.country_code}"],
    request_header_remove="referrer")

print(f"Ingress established at: {listener.url()}");
```
